	layerdataanalysis
	layerhardware
	threading
	common
	${PLAYERCPP_LINK_LIBS}

	boost_thread-mt

    ${LIBRARIES}
)

	layerautonomoustest
	layerdataanalysistest
	layerhardwaretest
	threadingtest
	commontest
